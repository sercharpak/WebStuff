function replaceAll( txt, busca, reemplaza ){
  var text=txt;
  while (text.toString().indexOf(busca) != -1)
      text = text.toString().replace(busca,reemplaza);
  return text;
}

function cambio_url(){
	url_fb = jQuery("#redes_sociales .fb-like iframe").attr("src");
	url_fb = replaceAll(url_fb, "%25","%");
	jQuery("#redes_sociales .fb-like iframe").attr("src",url_fb);
}
